let currentMode = null;

let pool = [];

let currentIndex = 0;

function shuffle(arr) {

  const a = [...arr];

  for (let i = a.length - 1; i > 0; i--) {

    const j = Math.floor(Math.random() * (i + 1));

    [a[i], a[j]] = [a[j], a[i]];

  }

  return a;

}

function startGame(mode) {

  currentMode = mode;

  pool = shuffle(questions[mode]);

  currentIndex = 0;

  document.getElementById('modeWrap').classList.add('hidden');

  document.getElementById('cardWrap').classList.remove('hidden');

  document.getElementById('bottomBar').style.display = 'flex';

  const labels = { light: 'ライト', standard: 'スタンダード', deep: 'ディープ' };

  document.getElementById('cardLabel').textContent = labels[mode];

  showQuestion();

}

function showQuestion() {

  const text = document.getElementById('questionText');

  const counter = document.getElementById('counter');

  const bar = document.getElementById('progressBar');

  counter.textContent = `${currentIndex + 1} / ${pool.length}`;

  bar.style.width = `${((currentIndex) / pool.length) * 100}%`;

  text.textContent = pool[currentIndex];

  text.classList.remove('exit', 'enter');

}

function nextQuestion(skip) {

  const text = document.getElementById('questionText');

  text.classList.add('exit');

  setTimeout(() => {

    currentIndex++;

    if (currentIndex >= pool.length) { showFinish(); return; }

    document.getElementById('progressBar').style.width = `${((currentIndex) / pool.length) * 100}%`;

    document.getElementById('counter').textContent = `${currentIndex + 1} / ${pool.length}`;

    text.textContent = pool[currentIndex];

    text.classList.remove('exit');

    text.classList.add('enter');

    setTimeout(() => text.classList.remove('enter'), 50);

  }, 280);

}

function showFinish() {

  document.getElementById('progressBar').style.width = '100%';

  document.getElementById('cardWrap').classList.add('hidden');

  document.getElementById('bottomBar').style.display = 'none';

  document.getElementById('counter').textContent = '';

  document.getElementById('finish').classList.add('visible');

}

function resetGame() {

  document.getElementById('finish').classList.remove('visible');

  document.getElementById('modeWrap').classList.remove('hidden');

  document.getElementById('progressBar').style.width = '0%';

}
